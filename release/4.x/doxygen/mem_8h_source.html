75 static bool is_vulnerable_to_stale_xapic_read()
76 {
77 CpuidInfo info;
78
79 cpuid(&info, 1, 0);
80
81 // Ignores stepping, looks only at model and family: potentially
82 // includes safe instances which differ only by stepping from a vulnerable
83 // instance.
84 constexpr uint64_t proc_id_mask = 0x000F'0FF0;
85 const uint64_t proc_id = info.eax & proc_id_mask;
86
87 // https://www.intel.com/content/www/us/en/developer/topic-technology/software-security-guidance/processors-affected-consolidated-product-cpu-model.html
88 // 2022 tab, column "Stale Data Read from Legacy xAPIC, CVE-2022-21233,
89 // INTEL-SA-00657"
90 const std::set<uint64_t> vulnerable_proc_ids{
91 0x506C0, // Apollo Lake
92 0x506F0, // Denverton (Goldmont)
93 0x606A0, // Ice Lake Xeon-SP
94 0x606C0, // Ice Lake D
95 0x706A0, // Gemini Lake
96 0x706E0, // Ice Lake U, Y
97 0x80660, // Snow Ridge BTS (Tremont)
98 0x806A0, // Lakefield B-step (Tremont)
99 0x806C0, // Tiger Lake U
100 0x806D0, // Tiger Lake H
101 0x90660, // Elkhart Lake (Tremont)
102 0x90670, // Alder Lake S (Golden Cove, Gracemont)
103 0x906A0, // Alder Lake H (Golden Cove, Gracemont)
104 0x906C0, // Jasper Lake (Tremont)
105 0xA0670 // Rocket Lake
106 };
107
108 const auto it = vulnerable_proc_ids.find(proc_id);
109 return it != vulnerable_proc_ids.end();
110 }
