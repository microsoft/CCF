48 struct TxID

55

57 // conversion

59
60 operator ccf::TxID() const
61 {
62 return {term, version};
63 }
64

66 {
67 return term == other.term && version == other.version;

71 {

73 }
74 };

291 constexpr auto public_domain_prefix = "public:";
292
293 static inline SecurityDomain get_security_domain(const std::string& name)
294 {
295 if (name.starts_with(public_domain_prefix))
296 {
297 return SecurityDomain::PUBLIC;
298 }
299
300 return SecurityDomain::PRIVATE;
301 }
302
303 static inline std::pair<SecurityDomain, AccessCategory> parse_map_name(
304 std::string_view name)
305 {
306 constexpr auto internal_category_prefix = "ccf.internal.";
307 constexpr auto governance_category_prefix = "ccf.gov.";
308 constexpr auto reserved_category_prefix = "ccf.";
309
310 auto security_domain = SecurityDomain::PRIVATE;
311 if (name.starts_with(public_domain_prefix))
312 {
313 name.remove_prefix(strlen(public_domain_prefix));
314 security_domain = SecurityDomain::PUBLIC;
315 }
316
317 auto access_category = AccessCategory::APPLICATION;
318 if (name.starts_with(internal_category_prefix))
319 {
320 access_category = AccessCategory::INTERNAL;
321 }
322 else if (name.starts_with(governance_category_prefix))
323 {
324 access_category = AccessCategory::GOVERNANCE;
325 }
326 else if (name.starts_with(reserved_category_prefix))
327 {
328 throw std::logic_error(fmt::format(
329 "Map name '{}' includes disallowed reserved prefix '{}'",
330 name,
331 reserved_category_prefix));
332 }
333
334 return {security_domain, access_category};
335 }

665 // shortly. In BFT mode when we deserialize and realize we need to
