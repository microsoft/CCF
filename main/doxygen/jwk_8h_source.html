54 static JsonWebKeyECCurve curve_id_to_jwk_curve(CurveID curve_id)
55 {
56 switch (curve_id)
57 {
58 case CurveID::SECP384R1:
59 return JsonWebKeyECCurve::P384;
60 case CurveID::SECP256R1:
61 return JsonWebKeyECCurve::P256;
62 case CurveID::SECP256K1:
63 return JsonWebKeyECCurve::P256K1;
64 default:
65 throw std::logic_error(fmt::format("Unknown curve {}", curve_id));
66 }
67 }
