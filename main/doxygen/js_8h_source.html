85 static inline void instantiate_authn_policies(
86 ccf::endpoints::EndpointDefinition& endpoint)
87 {
88 for (const auto& policy_desc : endpoint.properties.authn_policies)
89 {
90 if (policy_desc.is_string())
91 {
92 const auto policy_name = policy_desc.get<std::string>();
93 auto policy = get_policy_by_name(policy_name);
94 if (policy == nullptr)
95 {
96 throw std::logic_error(
97 fmt::format("Unknown auth policy: {}", policy_name));
98 }
99 endpoint.authn_policies.push_back(std::move(policy));
100 }
101 else
102 {
103 if (policy_desc.is_object())
104 {
105 const auto it = policy_desc.find("all_of");
106 if (it != policy_desc.end())
107 {
108 if (it.value().is_array())
109 {
110 std::vector<std::shared_ptr<ccf::AuthnPolicy>>
111 constituent_policies;
112 for (const auto& val : it.value())
113 {
114 if (!val.is_string())
115 {
116 constituent_policies.clear();
117 break;
118 }
119
120 const auto policy_name = val.get<std::string>();
121 auto policy = get_policy_by_name(policy_name);
122 if (policy == nullptr)
123 {
124 throw std::logic_error(
125 fmt::format("Unknown auth policy: {}", policy_name));
126 }
127 constituent_policies.push_back(std::move(policy));
128 }
129
130 if (!constituent_policies.empty())
131 {
132 endpoint.authn_policies.push_back(
133 std::make_shared<ccf::AllOfAuthnPolicy>(
134 constituent_policies));
135 continue;
136 }
137 }
138 }
139 }
140
141 // Any failure in above checks falls through to this detailed error.
142 throw std::logic_error(fmt::format(
143 "Unsupported auth policy. Policies must be either a string, or an "
144 "object containing an \"all_of\" key with list-of-strings value. "
145 "Unsupported value: {}",
146 policy_desc.dump()));
147 }
148 }
149 }
