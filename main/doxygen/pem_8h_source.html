131 static std::vector<ccf::crypto::Pem> split_x509_cert_bundle(
132 const std::string_view& pem)
133 {
134 std::string separator("-----END CERTIFICATE-----");
135 std::vector<ccf::crypto::Pem> pems;
136 auto separator_end = 0;
137 auto next_separator_start = pem.find(separator);
138 while (next_separator_start != std::string_view::npos)
139 {
140 pems.emplace_back(std::string(
141 pem.substr(separator_end, next_separator_start + separator.size())));
142 separator_end = next_separator_start + separator.size();
143 next_separator_start = pem.find(separator, separator_end);
144 }
145 return pems;
146 }
