68 bool inline matches_uvm_roots_of_trust(
69 const UVMEndorsements& endorsements,
70 const std::vector<UVMEndorsements>& uvm_roots_of_trust)
71 {
72 for (const auto& uvm_root_of_trust : uvm_roots_of_trust)
73 {
74 if (
75 uvm_root_of_trust.did == endorsements.did &&
76 uvm_root_of_trust.feed == endorsements.feed &&
77 uvm_root_of_trust.svn <= endorsements.svn)
78 {
79 return true;
80 }
81 }
82 return false;
83 }

90 static std::vector<std::vector<uint8_t>> decode_x5chain(
91 QCBORDecodeContext& ctx, const QCBORItem& x5chain)
92 {
93 std::vector<std::vector<uint8_t>> parsed;
94
95 if (x5chain.uDataType == QCBOR_TYPE_ARRAY)
96 {
97 QCBORDecode_EnterArrayFromMapN(&ctx, headers::PARAM_X5CHAIN);
98 while (true)
99 {
100 QCBORItem item;
101 auto result = QCBORDecode_GetNext(&ctx, &item);
102 if (result == QCBOR_ERR_NO_MORE_ITEMS)
103 {
104 break;
105 }
106 if (result != QCBOR_SUCCESS)
107 {
108 throw COSEDecodeError("Item in x5chain is not well-formed");
109 }
110 if (item.uDataType == QCBOR_TYPE_BYTE_STRING)
111 {
112 parsed.push_back(qcbor_buf_to_byte_vector(item.val.string));
113 }
114 else
115 {
116 throw COSEDecodeError(
117 "Next item in x5chain was not of type byte string");
118 }
119 }
120 QCBORDecode_ExitArray(&ctx);
121 if (parsed.empty())
122 {
123 throw COSEDecodeError("x5chain array length was 0 in COSE header");
124 }
125 }
126 else if (x5chain.uDataType == QCBOR_TYPE_BYTE_STRING)
127 {
128 parsed.push_back(qcbor_buf_to_byte_vector(x5chain.val.string));
129 }
130 else
131 {
132 throw COSEDecodeError(fmt::format(
133 "Value type {} of x5chain in COSE header is not array or byte string",
134 x5chain.uDataType));
135 }
136
137 return parsed;
138 }

140 static UvmEndorsementsProtectedHeader decode_protected_header(
141 const std::vector<uint8_t>& uvm_endorsements_raw)
142 {
143 UsefulBufC msg{uvm_endorsements_raw.data(), uvm_endorsements_raw.size()};
144
145 QCBORError qcbor_result;
146
147 QCBORDecodeContext ctx;
148 QCBORDecode_Init(&ctx, msg, QCBOR_DECODE_MODE_NORMAL);
149
150 QCBORDecode_EnterArray(&ctx, nullptr);
151 qcbor_result = QCBORDecode_GetError(&ctx);
152 if (qcbor_result != QCBOR_SUCCESS)
153 {
154 throw COSEDecodeError("Failed to parse COSE_Sign1 outer array");
155 }
156
157 uint64_t tag = QCBORDecode_GetNthTagOfLast(&ctx, 0);
158 if (tag != CBOR_TAG_COSE_SIGN1)
159 {
160 throw COSEDecodeError("Failed to parse COSE_Sign1 tag");
161 }
162
163 struct q_useful_buf_c protected_parameters;
164 QCBORDecode_EnterBstrWrapped(
165 &ctx, QCBOR_TAG_REQUIREMENT_NOT_A_TAG, &protected_parameters);
166 QCBORDecode_EnterMap(&ctx, NULL);
167
168 enum
169 {
170 ALG_INDEX,
171 CONTENT_TYPE_INDEX,
172 X5_CHAIN_INDEX,
173 ISS_INDEX,
174 FEED_INDEX,
175 END_INDEX
176 };
177 QCBORItem header_items[END_INDEX + 1];
178
179 header_items[ALG_INDEX].label.int64 = headers::PARAM_ALG;
180 header_items[ALG_INDEX].uLabelType = QCBOR_TYPE_INT64;
181 header_items[ALG_INDEX].uDataType = QCBOR_TYPE_INT64;
182
183 header_items[CONTENT_TYPE_INDEX].label.int64 =
184 headers::PARAM_CONTENT_TYPE;
185 header_items[CONTENT_TYPE_INDEX].uLabelType = QCBOR_TYPE_INT64;
186 header_items[CONTENT_TYPE_INDEX].uDataType = QCBOR_TYPE_TEXT_STRING;
187
188 header_items[X5_CHAIN_INDEX].label.int64 = headers::PARAM_X5CHAIN;
189 header_items[X5_CHAIN_INDEX].uLabelType = QCBOR_TYPE_INT64;
190 header_items[X5_CHAIN_INDEX].uDataType = QCBOR_TYPE_ANY;
191
192 header_items[ISS_INDEX].label.string =
193 UsefulBuf_FromSZ(HEADER_PARAM_ISSUER);
194 header_items[ISS_INDEX].uLabelType = QCBOR_TYPE_TEXT_STRING;
195 header_items[ISS_INDEX].uDataType = QCBOR_TYPE_TEXT_STRING;
196
197 header_items[FEED_INDEX].label.string =
198 UsefulBuf_FromSZ(HEADER_PARAM_FEED);
199 header_items[FEED_INDEX].uLabelType = QCBOR_TYPE_TEXT_STRING;
200 header_items[FEED_INDEX].uDataType = QCBOR_TYPE_TEXT_STRING;
201
202 header_items[END_INDEX].uLabelType = QCBOR_TYPE_NONE;
203
204 QCBORDecode_GetItemsInMap(&ctx, header_items);
205 qcbor_result = QCBORDecode_GetError(&ctx);
206 if (qcbor_result != QCBOR_SUCCESS)
207 {
208 throw COSEDecodeError("Failed to decode protected header");
209 }
210
211 UvmEndorsementsProtectedHeader phdr = {};
212
213 if (header_items[ALG_INDEX].uDataType != QCBOR_TYPE_NONE)
214 {
215 phdr.alg = header_items[ALG_INDEX].val.int64;
216 }
217
218 if (header_items[CONTENT_TYPE_INDEX].uDataType != QCBOR_TYPE_NONE)
219 {
220 phdr.content_type =
221 qcbor_buf_to_string(header_items[CONTENT_TYPE_INDEX].val.string);
222 }
223
224 if (header_items[X5_CHAIN_INDEX].uDataType != QCBOR_TYPE_NONE)
225 {
226 phdr.x5_chain = decode_x5chain(ctx, header_items[X5_CHAIN_INDEX]);
227 }
228
229 if (header_items[ISS_INDEX].uDataType != QCBOR_TYPE_NONE)
230 {
231 phdr.iss = qcbor_buf_to_string(header_items[ISS_INDEX].val.string);
232 }
233
234 if (header_items[FEED_INDEX].uDataType != QCBOR_TYPE_NONE)
235 {
236 phdr.feed = qcbor_buf_to_string(header_items[FEED_INDEX].val.string);
237 }
238
239 QCBORDecode_ExitMap(&ctx);
240 QCBORDecode_ExitBstrWrapped(&ctx);
241
242 qcbor_result = QCBORDecode_GetError(&ctx);
243 if (qcbor_result != QCBOR_SUCCESS)
244 {
245 throw COSEDecodeError(
246 fmt::format("Failed to decode protected header: {}", qcbor_result));
247 }
248
249 return phdr;
250 }

253 static std::span<const uint8_t> verify_uvm_endorsements_signature(
254 const ccf::crypto::Pem& leaf_cert_pub_key,
255 const std::vector<uint8_t>& uvm_endorsements_raw)
256 {
257 auto verifier = ccf::crypto::make_cose_verifier_from_key(leaf_cert_pub_key);
258
259 std::span<uint8_t> payload;
260 if (!verifier->verify(uvm_endorsements_raw, payload))
261 {
262 throw cose::COSESignatureValidationError("Signature verification failed");
263 }
264
265 return payload;

268 static UVMEndorsements verify_uvm_endorsements(
269 const std::vector<uint8_t>& uvm_endorsements_raw,
270 const pal::PlatformAttestationMeasurement& uvm_measurement,
271 const std::vector<UVMEndorsements>& uvm_roots_of_trust =
272 default_uvm_roots_of_trust)
273 {
274 auto phdr = cose::decode_protected_header(uvm_endorsements_raw);
275
276 if (!(cose::is_rsa_alg(phdr.alg) || cose::is_ecdsa_alg(phdr.alg)))
277 {
278 throw std::logic_error(fmt::format(
279 "Signature algorithm {} is not one of expected: RSA, ECDSA", phdr.alg));
280 }
281
282 std::string pem_chain;
283 for (auto const& c : phdr.x5_chain)
284 {
285 pem_chain += ccf::crypto::cert_der_to_pem(c).str();
286 }
287
288 const auto& did = phdr.iss;
289
290 auto did_document_str = didx509::resolve(pem_chain, did);
291 did::DIDDocument did_document = nlohmann::json::parse(did_document_str);
292
293 if (did_document.verification_method.empty())
294 {
295 throw std::logic_error(fmt::format(
296 "Could not find verification method for DID document: {}",
297 did_document_str));
298 }
299
300 ccf::crypto::Pem pubk;
301 for (auto const& vm : did_document.verification_method)
302 {
303 if (vm.controller == did && vm.public_key_jwk.has_value())
304 {
305 auto jwk = vm.public_key_jwk.value().get<ccf::crypto::JsonWebKey>();
306 switch (jwk.kty)
307 {
308 case ccf::crypto::JsonWebKeyType::RSA:
309 {
310 auto rsa_jwk =
311 vm.public_key_jwk->get<ccf::crypto::JsonWebKeyRSAPublic>();
312 pubk = ccf::crypto::make_rsa_public_key(rsa_jwk)->public_key_pem();
313 break;
314 }
315 case ccf::crypto::JsonWebKeyType::EC:
316 {
317 auto ec_jwk =
318 vm.public_key_jwk->get<ccf::crypto::JsonWebKeyECPublic>();
319 pubk = ccf::crypto::make_public_key(ec_jwk)->public_key_pem();
320 break;
321 }
322 default:
323 {
324 throw std::logic_error(fmt::format(
325 "Unsupported public key type ({}) for DID {}", jwk.kty, did));
326 }
327 }
328 }
329 }
330
331 if (pubk.empty())
332 {
333 throw std::logic_error(fmt::format(
334 "Could not find matching public key for DID {} in {}",
335 did,
336 did_document_str));
337 }
338
339 auto raw_payload =
340 verify_uvm_endorsements_signature(pubk, uvm_endorsements_raw);
341
342 if (phdr.content_type != cose::headers::CONTENT_TYPE_APPLICATION_JSON_VALUE)
343 {
344 throw std::logic_error(fmt::format(
345 "Unexpected payload content type {}, expected {}",
346 phdr.content_type,
347 cose::headers::CONTENT_TYPE_APPLICATION_JSON_VALUE));
348 }
349
350 UVMEndorsementsPayload payload = nlohmann::json::parse(raw_payload);
351 if (payload.sevsnpvm_launch_measurement != uvm_measurement.hex_str())
352 {
353 throw std::logic_error(fmt::format(
354 "Launch measurement in UVM endorsements payload {} is not equal "
355 "to UVM attestation measurement {}",
356 payload.sevsnpvm_launch_measurement,
357 uvm_measurement.hex_str()));
358 }
359
360 LOG_INFO_FMT(
361 "Successfully verified endorsements for attested measurement {} against "
362 "{}, feed {}, svn {}",
363 payload.sevsnpvm_launch_measurement,
364 did,
365 phdr.feed,
366 payload.sevsnpvm_guest_svn);
367
368 UVMEndorsements end{did, phdr.feed, payload.sevsnpvm_guest_svn};
369
370 if (!matches_uvm_roots_of_trust(end, uvm_roots_of_trust))
371 {
372 throw std::logic_error(fmt::format(
373 "UVM endorsements did {}, feed {}, svn {} "
374 "do not match any of the known UVM roots of trust",
375 end.did,
376 end.feed,
377 end.svn));
378 }
379
380 return end;
381 }
