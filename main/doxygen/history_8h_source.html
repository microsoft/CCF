230 // Use optimised CCF openssl_sha256 function to avoid performance regression
231 // on OpenSSL 3.x
232 static constexpr size_t sha256_byte_size = 32;
233 static inline void sha256_history(
234 const merkle::HashT<sha256_byte_size>& l,
235 const merkle::HashT<sha256_byte_size>& r,
236 merkle::HashT<sha256_byte_size>& out)
237
238 {
239 uint8_t block[sha256_byte_size * 2];
240 memcpy(&block[0], l.bytes, sha256_byte_size);
241 memcpy(&block[sha256_byte_size], r.bytes, sha256_byte_size);
242
243 ccf::crypto::openssl_sha256(block, out.bytes);
244 }

248 class Proof
249 {
250 private:
251 HistoryTree::Hash root;
252 std::shared_ptr<HistoryTree::Path> path = nullptr;
253
254 public:
255 Proof() {}
256
257 Proof(const std::vector<uint8_t>& v)
258 {
259 size_t position = 0;
260 root.deserialise(v, position);
261 path = std::make_shared<HistoryTree::Path>(v, position);
262 }
263
264 const HistoryTree::Hash& get_root() const
265 {
266 return root;
267 }
268
269 std::shared_ptr<HistoryTree::Path> get_path()
270 {
271 return path;
272 }
273
274 Proof(HistoryTree* tree, uint64_t index)
275 {
276 root = tree->root();
277 path = tree->path(index);
278 }
279
280 Proof(const Proof&) = delete;
281
282 bool verify(HistoryTree* tree) const
283 {
284 if (path->max_index() > tree->max_index())
285 {
286 return false;
287 }
288 else if (tree->max_index() == path->max_index())
289 {
290 return tree->root() == root && path->verify(root);
291 }
292 else
293 {
294 auto past_root = tree->past_root(path->max_index());
295 return path->verify(*past_root);
296 }
297 }
298
299 std::vector<uint8_t> to_v() const
300 {
301 std::vector<uint8_t> v;
302 root.serialise(v);
303 path->serialise(v);
304 return v;
305 }
306 };

531
532 void start_signature_emit_timer() override
533 {
534 struct EmitSigMsg
535 {
536 EmitSigMsg(HashedTxHistory<T>* self_) : self(self_) {}
537 HashedTxHistory<T>* self;
538 };
539
540 auto emit_sig_msg = std::make_unique<::threading::Tmsg<EmitSigMsg>>(
541 [](std::unique_ptr<::threading::Tmsg<EmitSigMsg>> msg) {
542 auto self = msg->data.self;
543
544 std::unique_lock<ccf::pal::Mutex> mguard(
545 self->signature_lock, std::defer_lock);
546
547 bool should_emit_signature = false;
548
549 if (mguard.try_lock())
550 {
551 auto consensus = self->store.get_consensus();
552 if (consensus != nullptr)
553 {
554 auto sig_disp = consensus->get_signature_disposition();
555 switch (sig_disp)
556 {
557 case ccf::kv::Consensus::SignatureDisposition::CANT_REPLICATE:
558 {
559 break;
560 }
561 case ccf::kv::Consensus::SignatureDisposition::CAN_SIGN:
562 {
563 if (self->store.committable_gap() > 0)
564 {
565 should_emit_signature = true;
566 }
567 break;
568 }
569 case ccf::kv::Consensus::SignatureDisposition::SHOULD_SIGN:
570 {
571 should_emit_signature = true;
572 break;
573 }
574 }
575 }
576 }
577
578 if (should_emit_signature)
579 {
580 msg->data.self->emit_signature();
581 }
582
583 self->emit_signature_timer_entry =
584 ::threading::ThreadMessaging::instance().add_task_after(
585 std::move(msg), std::chrono::milliseconds(self->sig_ms_interval));
586 },
587 this);
588
589 emit_signature_timer_entry =
590 ::threading::ThreadMessaging::instance().add_task_after(
591 std::move(emit_sig_msg), std::chrono::milliseconds(sig_ms_interval));
592 }
593
594 ~HashedTxHistory()
595 {
596 if (emit_signature_timer_entry.has_value())
597 {
598 ::threading::ThreadMessaging::instance().cancel_timer_task(
599 *emit_signature_timer_entry);
600 }
601 }

608 bool init_from_snapshot(
609 const std::vector<uint8_t>& hash_at_snapshot) override
610 {
611 std::lock_guard<ccf::pal::Mutex> guard(state_lock);
612 // The history can be initialised after a snapshot has been applied by
613 // deserialising the tree in the signatures table and then applying the
614 // hash of the transaction at which the snapshot was taken
615 auto tx = store.create_read_only_tx();
616 auto tree_h = tx.template ro<ccf::SerialisedMerkleTree>(
617 ccf::Tables::SERIALISED_MERKLE_TREE);
618 auto tree = tree_h->get();
619 if (!tree.has_value())
620 {
621 LOG_FAIL_FMT("No tree found in serialised tree map");
622 return false;
623 }
624
625 CCF_ASSERT_FMT(
626 !replicated_state_tree.in_range(1),
627 "Tree is not empty before initialising from snapshot");
628
629 replicated_state_tree.deserialise(tree.value());
630
631 ccf::crypto::Sha256Hash hash;
632 std::copy_n(
633 hash_at_snapshot.begin(),
634 ccf::crypto::Sha256Hash::SIZE,
635 hash.h.begin());
636 replicated_state_tree.append(hash);
637 return true;
638 }

657 ccf::kv::TxHistory::Result verify_and_sign(
658 PrimarySignature& sig,
659 ccf::kv::Term* term,
660 ccf::kv::Configuration::Nodes& config) override
661 {
662 if (!verify(term, &sig))
663 {
664 return ccf::kv::TxHistory::Result::FAIL;
665 }
666
667 ccf::kv::TxHistory::Result result = ccf::kv::TxHistory::Result::OK;
668

675 bool verify(
676 ccf::kv::Term* term = nullptr,
677 PrimarySignature* signature = nullptr) override
678 {
679 auto tx = store.create_read_only_tx();
680 auto signatures =
681 tx.template ro<ccf::Signatures>(ccf::Tables::SIGNATURES);
682 auto sig = signatures->get();
683 if (!sig.has_value())
684 {
685 LOG_FAIL_FMT("No signature found in signatures map");
686 return false;
687 }
688 auto& sig_value = sig.value();
689 if (term)
690 {
691 *term = sig_value.view;
692 }
693
694 if (signature)
695 {
696 *signature = sig_value;
697 }
698
699 auto root = get_replicated_state_root();
700 log_hash(root, VERIFY);
701 return verify_node_signature(tx, sig_value.node, sig_value.sig, root);
702 }

751
752 void try_emit_signature() override
753 {
754 std::unique_lock<ccf::pal::Mutex> mguard(signature_lock, std::defer_lock);
755 if (store.committable_gap() < sig_tx_interval || !mguard.try_lock())
756 {
757 return;
758 }
759
760 if (store.committable_gap() >= sig_tx_interval)
761 {
762 mguard.unlock();
763 emit_signature();
764 }

767 void emit_signature() override
768 {
769 // Signatures are only emitted when there is a consensus
770 auto consensus = store.get_consensus();
771 if (!consensus)
772 {
773 return;
774 }
775
776 if (!endorsed_cert.has_value())
777 {
778 throw std::logic_error(
779 fmt::format("No endorsed certificate set to emit signature"));
780 }
781
782 auto txid = store.next_txid();
783
784 LOG_DEBUG_FMT("Signed at {} in view: {}", txid.version, txid.term);
785
786 store.commit(
787 txid,
788 std::make_unique<MerkleTreeHistoryPendingTx<T>>(
789 txid, store, *this, id, kp, endorsed_cert.value()),
790 true);
791 }
