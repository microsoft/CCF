55 Write typed_writes;
56 for (const auto& [uk, opt_uv] : w)
57 {
58 if (!opt_uv.has_value())
59 {
60 // Deletions are indicated by nullopt. We cannot deserialise them,
61 // they are deletions here as well

63 }
